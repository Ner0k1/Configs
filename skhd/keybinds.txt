# switching window focus
alt + shift - h -> focus west
alt + shift - l -> focus east

# change focus between spaces
alt + shift + cmd - 1 -> space 1
alt + shift + cmd - 2 -> space 2
alt + shift + cmd - 3 -> space 3 
# rearrange current layout
alt - r -> rotate 270
alt - y -> axis
alt - x -> axis

# change size of windows 
alt + shift - m -> fullscreen
alt + shift - e -> balance

# quick open apps
ctrl + fn - b -> open Waterfox.app
ctrl + fn - c -> open Alacritty.app
ctrl + fn - m -> open Prism Launcher.app
ctrl + fn - o -> open OpenOffice.app
ctrl + fn - t -> open Telegram.app
ctrl + fn - w -> open WhatsApp.app
ctrl + fn - n -> open Joplin.app
ctrl + fn - i -> open IntelliJ IDEA Ultimate.app
ctrl + fn - a -> open "About This Mac"
ctrl + shift + fn - s -> open "x-apple.systempreferences:" 

# switch to stack mode
ctrl + shift - space -> layout stack
alt + shift - space -> layout bsp

# move between windows in stack
alt - h-> focus stack.prev
alt - l-> focus stack.next
alt - j-> focus stack.first
alt - k-> focus stack.last
